
function bubble_sort: (list) -> ();
bubble_sort { list arr } = {
	println("called bubble sort");
	int n;
	n = list_size(arr);
	int i; int j; int tmp;
	i = 0;
	<loop:{
		<if:i >= n:{break;}:>
		j = 0;
		<loop:{
			<if: list_get(arr, j) > list_get(arr, j+1):{
				tmp = list_get(arr, j);
				list_set(arr, j, list_get(arr, j+1));
				list_set(arr, j+1, tmp);
				
				println(arr);
			}:>

			j = j + 1;
			<if:j >= n - i - 1:{break;}:>
		}:>
		i = i + 1;
	}:>
};
